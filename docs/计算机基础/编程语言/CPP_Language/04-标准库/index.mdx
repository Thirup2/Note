## 一、IO库

```c++
// 头文件
#include <iostream>
// 类型
using std::istream;
using std::ostream;
using std::iostream;
// 对象
using std::cin;
using std::cout;
using std::cerr;
using std::clog;
using std::endl;
```



## 二、string类

```c++
// 头文件
#include <string>
#include <cctype>		// 字符处理函数库(char)
#include <cwctype>		// 字符处理函数库(wchar_t)
// 类型
using std::string;
using std::wstring;
using std::u16string;
using std::u32string;
// 函数
using std::getline;
```
