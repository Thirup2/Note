```c
struct items
{
    int x;
    int y;
};
typedef struct items ElemType;

struct position
{
    int level;						// 深度
    int order;						// 当前深度的位序
};									// 定位类型
typedef struct position Pos;		// 用于定位

struct status
{
    ElemType elem;
    POS position;
};									// 保存当前结点所有信息
typedef struct status Status;		// 状态元素类型

struct node
{
    Status data;					// 包含状态的数据域
    struct node *leftchild;			// 左孩子
    struct node *rightchild;		// 右孩子
    struct node *parent;			// 双亲
    struct node *rightbrother;		// 右兄弟
};
typedef struct node Node;			// 结点类型
typedef Node *PNode;				// 结点指针

struct tree
{
    PNode root;						// 头指针
    int length;						// 当前树的结点个数
};
typedef struct tree BiTree;			// 链式二叉树
typedef BiTree PtrBiTree;			// 链式二叉树指针
```
