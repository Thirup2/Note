1. 代码实现 

#define MAXSIZE 20
struct items
{
    int x;
    int y;
};                                // 数据项定义
typedef struct items ElemType;    // 数据元素类型

struct queue
{
    ElemType data[MAXSIZE];
    int front;
    int rear;
};                                // 数据类型定义
typedef struct queue Queue;        // 队列类型
