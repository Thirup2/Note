```c
struct items
{
    int x;                        // 数据项1类型
    int y;                        // 数据项2类型
};
typedef struct items ElemType;    // 数据元素类型
struct node
{
    ElemType data;
    struct node *next;
    struct node *pre;
};
typedef struct node Node;         // 数据对象类型
typedef Node *PNode;
struct list
{
    PNode start;
    int length;
};
typedef struct list DulCycleList; // 双向循环链表类型
```
